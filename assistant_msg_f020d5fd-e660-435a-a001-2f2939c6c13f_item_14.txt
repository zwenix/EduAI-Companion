### 3️⃣ Dynamic Prompt Assembly System

```typescript
// src/lib/prompt-engine.ts
import { ENHANCED_MASTER_PROMPT } from './prompts/master-prompt';
import { 
  WORKSHEET_PROMPT_TEMPLATE, 
  VISUAL_AID_PROMPT_TEMPLATE, 
  STUDY_GUIDE_PROMPT_TEMPLATE 
} from './prompts/content-templates';

export interface PromptContext {
  contentType: 'worksheet' | 'poster' | 'study-guide' | 'infographic' | 'admin-doc';
  grade: string;
  subject: string;
  topic: string;
  language: string;
  learnerProfile?: string;
  additionalInstructions?: string;
  visualStyle?: 'modern' | 'playful' | 'professional' | 'minimalist';
  colorScheme?: string;
  capsReference?: string;
}

export class EduAIPromptEngine {
  
  /**
   * Assemble a complete, context-aware prompt for any content type
   */
  static assemblePrompt(context: PromptContext): { system: string; user: string } {
    const phase = this.getPhaseByGrade(context.grade);
    const subjectPalette = this.getSubjectPalette(context.subject);
    
    // Enhance image prompt with context
    const enhancedImagePrompt = this.enhanceImagePrompt(
      context.topic, 
      context.grade, 
      context.subject,
      phase,
      subjectPalette
    );
    
    // Select base template
    let contentTemplate = '';
    switch (context.contentType) {
      case 'worksheet':
        contentTemplate = WORKSHEET_PROMPT_TEMPLATE;
        break;
      case 'poster':
      case 'infographic':
        contentTemplate = VISUAL_AID_PROMPT_TEMPLATE;
        break;
      case 'study-guide':
        contentTemplate = STUDY_GUIDE_PROMPT_TEMPLATE;
        break;
      default:
        contentTemplate = WORKSHEET_PROMPT_TEMPLATE; // fallback
    }
    
    // Inject dynamic values into template
    const userPrompt = contentTemplate
      .replace(/\$\{topic\}/g, context.topic)
      .replace(/\$\{grade\}/g, context.grade)
      .replace(/\$\{subject\}/g, context.subject)
      .replace(/\$\{capsCode\}/g, context.capsReference || 'TBD')
      .replace(/\$\{subject-color\}/g, subjectPalette.primary)
      .replace(/\$\{subject-dark\}/g, subjectPalette.dark)
      .replace(/\$\{subject-light\}/g, subjectPalette.light)
      .replace(/\$\{accent-color\}/g, subjectPalette.accent)
      .replace(/\$\{imagePrompt\}/g, enhancedImagePrompt)
      .replace(/\$\{phase\}/g, phase);
    
    // Enhance system prompt with phase-specific guidance
    const systemPrompt = ENHANCED_MASTER_PROMPT
      .replace(/\$\{phase\}/g, phase)
      .replace(/\$\{gradeRange\}/g, this.getGradeRange(phase));
    
    return {
      system: systemPrompt,
      user: userPrompt
    };
  }
  
  /**
   * Get educational phase from grade
   */
  private static getPhaseByGrade(grade: string): string {
    const numGrade = parseInt(grade);
    if (grade === 'R' || numGrade <= 3) return 'Foundation Phase';
    if (numGrade <= 6) return 'Intermediate Phase';
    if (numGrade <= 9) return 'Senior Phase';
    return 'FET Phase';
  }
  
  private static getGradeRange(phase: string): string {
    const ranges: Record<string, string> = {
      'Foundation Phase': 'Grade R-3',
      'Intermediate Phase': 'Grade 4-6', 
      'Senior Phase': 'Grade 7-9',
      'FET Phase': 'Grade 10-12'
    };
    return ranges[phase] || 'Grade R-12';
  }
  
  /**
   * Get subject-aligned color palette
   */
  private static getSubjectPalette(subject: string): {
    primary: string; dark: string; light: string; accent: string;
  } {
    const palettes: Record<string, any> = {
      'Mathematics': { primary: '#2563eb', dark: '#1e40af', light: '#dbeafe', accent: '#60a5fa' },
      'Mathematical Literacy': { primary: '#2563eb', dark: '#1e40af', light: '#dbeafe', accent: '#60a5fa' },
      'English': { primary: '#7c3aed', dark: '#5b21b6', light: '#ede9fe', accent: '#a78bfa' },
      'Home Language': { primary: '#7c3aed', dark: '#5b21b6', light: '#ede9fe', accent: '#a78bfa' },

      'Life Orientation': { primary: '#f97316', dark: '#c2410c', light: '#ffedd5', accent: '#fbbf24' },
      'Natural Sciences': { primary: '#059669', dark: '#047857', light: '#d1fae5', accent: '#34d399' },

      'Economic Management Sciences': { primary: '#f59e0b', dark: '#b45309', light: '#fef3c7', accent: '#fbbf24' },
```

### 5️⃣ Quality Assurance & Testing Protocol

Add this to `src/lib/prompt-validator.ts`:

```typescript
export class PromptQualityValidator {
  
  static validateOutput(html: string, context: PromptContext): ValidationResult {
    const issues: string[] = [];
    const warnings: string[] = [];
    
    // ✅ Visual hierarchy checks
    if (!html.includes('hero-section') || !html.includes('banner')) {
      issues.push('Missing required visual hierarchy elements (hero/banner)');
    }
    
    // ✅ Child-appropriate font sizing
    const fontSizeMatch = html.match(/text-(\w+)/g);
    const hasSmallFonts = fontSizeMatch?.some(cls => 
      ['text-xs', 'text-[10px]', 'text-[11px]'].includes(cls)
    );
    if (hasSmallFonts && ['R','1','2','3'].includes(context.grade)) {
      warnings.push('Font size may be too small for Foundation Phase learners');
    }
    
    // ✅ Color contrast (basic heuristic)
    const hasSubjectColor = html.includes(context.subject?.toLowerCase()) || 
                           html.includes('blue-') || html.includes('green-') || 
                           html.includes('purple-') || html.includes('orange-');
    if (!hasSubjectColor) {
      warnings.push('Subject-aligned color scheme not detected');
    }
    
    // ✅ Print readiness
    if (!html.includes('@media print') && !html.includes('print:')) {
      warnings.push('Print optimization styles may be missing');
    }
    
    // ✅ Accessibility basics
    if (!html.includes('alt=') && html.includes('<img')) {
      issues.push('Image elements missing alt text placeholders');
    }
    
    // ✅ CAPS alignment marker
    if (!html.includes('CAPS') && !html.includes('caps')) {
      warnings.push('CAPS alignment not explicitly mentioned in output');
    }
    
    return {
      isValid: issues.length === 0,
      issues,
      warnings,
      score: this.calculateScore(issues.length, warnings.length)
    };
  }
  
  private static calculateScore(errors: number, warnings: number): number {
    return Math.max(0, 100 - (errors * 25) - (warnings * 5));
  }
}

interface ValidationResult {
  isValid: boolean;
  issues: string[];
  warnings: string[];
  score: number; // 0-100
}
```
